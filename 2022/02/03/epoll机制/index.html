代码：

int createSocket(std::string ip, std::string port)
{
    int sock = socket(AF_INET, SOCK_STREAM, 0);
    if (sock < 0)
    {
        std::cerr << "socket创建失败" << std::endl;
        exit(2);
    }
    std::cout << "socket创建成功" << std::endl;
    // 用于设置监听句柄可重用设置
    int opt = 1;
    if (setsockopt(sock, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt)) < 0)
    {
        std::cerr << "地址重用设置失败" << std::endl;
    }

    sockaddr_in local;
    local.sin_family = AF_INET;
    local.sin_port = htons(atoi(port.c_str()));
    local.sin_addr.s_addr = inet_addr(ip.c_str());
    // 绑定，使sock套接字与创建的地址local相关联
    if (bind(sock, (sockaddr *)&local, sizeof(local)) < 0)
    {
        std::cout << "bind error" << std::endl;
        exit(2);
    }
    std::cout << "bind success" << std::endl;
    // listen函数使主动连接的sock变为等待用户连接的被动型
    if (listen(sock, 5) < 0)
    {
        std::cout << "listen error" << std::endl;
        exit(2);
    }
    std::cout << "listen success" << std::endl;
    return sock;
}

int setNoblock(int sock)
{
    // 获得文件状态
    int fileState = fcntl(sock, F_GETFL);
    // 通过位运算追加一个非阻塞的选项
    return fcntl(sock, F_SETFL, fileState | O_NONBLOCK);
}

int main(int argc, char const *argv[])
{
    // if (argc != 3)
    // {
    //     std::cout << "请输入ip和端口" << std::endl;
    //     exit(1);
    // }
    int listenSock = createSocket("0.0.0.0", "9090");
    // 第一步：创建epoll_fd对象
    // 生成epoll专用的文件描述符，理解为在内核申请一个空间，用来存放想关注的sock_fd上发生了什么事件
    int epollFd = epoll_create(256);
    if (epollFd < 0)
    {
        std::cout << "epollFd 创建失败" << std::endl;
        exit(1);
    }
    std::cout << "epollFd 创建成功" << std::endl;
    // 第二步：设置epoll_event对象
    epoll_event epollEvent;
    epollEvent.events = EPOLLIN;     //将当前句柄转化为一个输入事件
    epollEvent.data.fd = listenSock; //理解为与事件相关的文件的描述符
    // 添加关心的事件
    if (epoll_ctl(epollFd, EPOLL_CTL_ADD, listenSock, &epollEvent) < 0)
    {
        std::cout << "添加关心事件失败" << std::endl;
        exit(1);
    }
    // 定义用来保存就绪事件的数组
    epoll_event readyEvent[128];
    memset(readyEvent, 0, sizeof(readyEvent));
    int maxNum = 128;   // 就绪数组的最大值
    int timeOut = 1000; // 超时时间
    int ret = 0;        // 返回值
    while (1)
    {
        switch (ret = epoll_wait(epollFd, readyEvent, maxNum, timeOut))
        {
        case -1:
            std::cout << "epoll_wait......." << std::endl;
            break;
        case 0:
            std::cout << "time_out........." << std::endl;
            break;
        default:
        {
            // 至少有一个事件就绪
            int i = 0;
            for (; i < ret; ++i)
            {
                // 从wait阻塞等待时返回的ready_ev数组中的fd
                int fd = readyEvent[i].data.fd;
                // 判断fd是哪个句柄，是什么事件(fd为listenSock且是输入事件)
                if ((fd == listenSock) && (readyEvent[i].events == EPOLLIN))
                {
                    //第一次意味着有客户端连接进来
                    sockaddr_in remote;
                    socklen_t len = sizeof(remote);
                    int acceptSock = accept(listenSock, (sockaddr *)&remote, &len);
                    if (acceptSock < 0)
                    {
                        std::cout << "accept error" << std::endl;
                        continue;
                    }
                    printf("accept a client .. [ip]: %s,[port]:%d\n",
                           inet_ntoa(remote.sin_addr), ntohs(remote.sin_port));
                    // 将新的事件添加到epoll事件里
                    epollEvent.events = EPOLLIN | EPOLLET; //设置了边缘触发
                    epollEvent.data.fd = acceptSock;
                    setNoblock(acceptSock); // 把当前选项设置为非阻塞
                    if (epoll_ctl(epollFd, EPOLL_CTL_ADD, acceptSock, &epollEvent) < 0)
                    {
                        std::cout << "epoll_ctl 失败" << std::endl;
                        close(acceptSock);
                    }
                }
                else
                {
                    // 普通的IO(连接句柄的输入了)
                    if (readyEvent[i].events == EPOLLIN)
                    {
                        // 表示对端发送消息过来了
                        char buf[1024];
                        memset(buf, 0, sizeof(buf));
                        int ret = recv(fd, buf, sizeof(buf) - 1, 0);
                        if (ret < 0)
                        {
                            std::cout << "recv error" << std::endl;
                            continue;
                        }
                        else if (ret == 0)
                        {
                            std::cout << "remote close..." << std::endl;
                            epoll_ctl(epollFd, EPOLL_CTL_DEL, fd, NULL);
                            close(fd);
                        }
                        else
                        {
                            // 读取成功，输出数据
                            std::cout << "client : " << buf << std::endl;
                            // 把事件写入输出事件，输入事件完成了
                            epollEvent.data.fd = fd;
                            epollEvent.events = EPOLLOUT | EPOLLET;
                            epoll_ctl(epollFd, EPOLL_CTL_MOD, fd, &epollEvent);
                            std::cout << "事件修改成功" << std::endl;
                        }
                    }
                    else if (readyEvent[i].events == EPOLLOUT)
                    {
                        std::cout << "正在准备发送..." << std::endl;
                        const char *msg = "HTTP/1.1 200 OK \r\n\r\n<h1> hello message </h1>\r\n";
                        if (send(fd, msg, strlen(msg), 0) < 0)
                        {
                            std::cout << "发送失败" << std::endl;
                        }
                        epoll_ctl(epollFd, EPOLL_CTL_DEL, fd, NULL);
                        close(fd);
                    }
                }
            }
        }
        break;
        }
    }
    close(listenSock);
    return 0;
}
