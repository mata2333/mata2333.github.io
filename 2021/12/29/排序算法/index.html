void HalfInsertSort(int a[], int n)
{
	int i, j, low, high, mid, tmp;
	for (i = 1; i < n; i++)
	{
		tmp = a[i];
		low = 0; high = i - 1;
		while (low <= high)
		{
			mid = (low + high) / 2;
			if (a[mid] > tmp)high = mid - 1;
			else low = mid + 1;
		}
		for (j = i - 1; j >= high + 1; j--)
			a[j + 1] = a[j];
		a[high + 1] = tmp;
	}
}

void ShellSort(int a[], int n)
{
	int i, j, tmp, d;
	for (d = n / 2; d >= 1; d /= 2)
	{
		for (i = d; i < n; i++)
		{
			if (a[i] < a[i - d])
			{
				tmp = a[i];
				for (j = i - d; j >= 0 && a[j] > tmp ; j-=d)
				{
					a[j + d] = a[j];
				}
				a[j + d] = tmp;
			}
		}
	}
}

void InsertSort(int a[], int n)

	for (i = 1; i < n; i++)

		{

int Rand(int x, int y)
{
	int b, d;
	if (y > x)
	{
		b = x;
		d = y - x + 1;
	}
	else
	{
		b = y;
		d = x - y + 1;
	}
	return (int)(1.0 * rand() * d / RAND_MAX + b);
}

void HalfInsertSort(int a[], int n)
{
	int i, j, low, high, mid, tmp;
	for (i = 1; i < n; i++)
	{
		tmp = a[i];
		low = 0; high = i - 1;
		while (low <= high)
		{
			mid = (low + high) / 2;
			if (a[mid] > tmp)high = mid - 1;
			else low = mid + 1;
		}
		for (j = i - 1; j >= high + 1; j--)
			a[j + 1] = a[j];
		a[high + 1] = tmp;
	}
}

void ShellSort(int a[], int n)
{
	int i, j, tmp, d;
	for (d = n / 2; d >= 1; d /= 2)
	{
		for (i = d; i < n; i++)
		{
			if (a[i] < a[i - d])
			{
				tmp = a[i];
				for (j = i - d; j >= 0 && a[j] > tmp; j-=d)
				{
					a[j + d] = a[j];
				}
				a[j + d] = tmp;
			}
		}
	}
}

void BubbleSort(int a[], int n)
{
	int tmp;
	int flag;
	for (int i = 0; i < n - 1; i++)
	{
		flag = 0;
		for (int j = n - 1; j > i; j--)
		{
			if (a[j] < a[j - 1])
			{
				tmp = a[j - 1];
				a[j - 1] = a[j];
				a[j] = tmp;
				flag = 1;
			}
		}
		if (flag == 0)
			return;
	}
}

int Partition(int a[], int low, int high)
{
	int p = a[low];// ä½¿éœ€è¦æ’åºçš„æ•°ç»„çš„ç¬¬ä¸€ä¸ªå…ƒç´ ä½œä¸ºæ¢çº½
	while (low < high)
	{
		// ä¸€ç›´ä½¿highå‡1ï¼Œç›´åˆ°æ‰¾åˆ°ä¸€ä¸ªå€¼å¤§äºæ¢çº½
		while (low < high && a[high] >= p)high--;
		a[low] = a[high];// å°†æ‰¾åˆ°çš„å¤§äºæ¢çº½çš„å€¼èµ‹äºˆlowä½ç½®
		// åŒä¸Š
		while (low < high && a[low] <= p)low++;
		a[high] = a[low];
	}
	a[low] = p;
	return low;
}
void QuickSort(int a[], int low, int high)
{
	if (low < high)
	{
		int p = Partition(a, low, high);
		// å¯¹æ¢çº½å·¦è¾¹è¿›è¡Œå¿«é€Ÿæ’åº
		QuickSort(a, low, p - 1);
		// å¯¹æ¢çº½å³è¾¹è¿›è¡Œå¿«é€Ÿæ’åº
		QuickSort(a, p + 1, high);
	}
}

void SelectSort(int a[], int n)
{
	for (int i = 0; i < n; i++)
	{
		for (int j = i + 1; j < n; j++)
		{
			if (a[i] > a[j])
			{
				int tmp = a[i];
				a[i] = a[j];
				a[j] = tmp;
			}
		}
	}
}

void HeadAdjust(int a[], int k, int len)
{
	int tmp = a[k];
	// i=2*kæŒ‡å‘èŠ‚ç‚¹kçš„å·¦å­©å­
	// i*=2 æ„ä¸ºæ¯æ¬¡éƒ½é‡æ–°æŒ‡å‘æ–°çš„å­©å­èŠ‚ç‚¹
	for (int i = 2 * k; i <= len; i*=2)
	{
		if (i < len && a[i] < a[i + 1])
			i++;// ç­›é€‰å·¦å­©å­æˆ–è€…å³å­©å­å“ªä¸ªå€¼è¾ƒå¤§
		// æ­¤æ—¶a[i]æŒ‡å‘çš„èŠ‚ç‚¹ä¸ºä¸¤ä¸ªå­©å­ä¹‹é—´è¾ƒå¤§å€¼
		if (tmp >= a[i])break;// æ»¡è¶³å¤§æ ¹å †ï¼Œæ— éœ€è°ƒæ•´ï¼Œè·³å‡ºå¾ªç¯
		else
		{
			a[k] = a[i];// å°†æ ¹èŠ‚ç‚¹è®¾ç½®ä¸ºè¯¥å­èŠ‚ç‚¹çš„è¾ƒå¤§å€¼
			k = i;// æ›´æ–°æ ¹èŠ‚ç‚¹
		}
	}
	a[k] = tmp;// è¢«ç­›é€‰èŠ‚ç‚¹æ”¾å…¥æœ€ç»ˆä½ç½®
}
void BuildMaxHeap(int a[], int len)
{
	for (int i = len / 2; i > 0; i--)
		HeadAdjust(a, i, len);
}
void HeapSort(int a[], int len)
{
	BuildMaxHeap(a, len);
	int tmp;
	for (int i = len; i > 1; i--)
	{
		tmp = a[i];
		a[i] = a[1];
		a[1] = tmp;
		HeadAdjust(a, 1, i - 1);
	}
}
