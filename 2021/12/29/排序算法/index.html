插入排序

直接插入排序

// 直接插入排序
void InsertSort(int a[], int n)
{
	int i,j,temp; //temp是为了用来保存元素，防止原下标之前的元素在后移的过程中将该元素覆盖
	for (i = 1; i < n; i++)
	{
		if (a[i] < a[i - 1])// 如果当前元素比前一个元素要小，那么前面的元素就需要一直后移
		{
			int temp = a[i];
			for (j = i - 1; j >= 0; j--)
			{
				if (a[j] > temp)// 凡是大于temp的元素都需要后移
					a[j + 1] = a[j];
				else break;
			}
			a[j + 1] = temp;
		}
	}
}

折半插入排序

// 折半插入排序
void HalfInsertSort(int a[], int n)
{
	int i, j, low, high, mid, tmp;
	for (i = 1; i < n; i++)
	{
		tmp = a[i];
		low = 0; high = i - 1;
		while (low <= high)
		{
			mid = (low + high) / 2;
			if (a[mid] > tmp)high = mid - 1;
			else low = mid + 1;
		}
		for (j = i - 1; j >= high + 1; j--)
			a[j + 1] = a[j];
		a[high + 1] = tmp;
	}
}

折半插入排序其实也只是将向前查找比tmp小的值的方式换成了折半查找，但是移动的时候还是需要移动n次，所以本质上的时间复杂度仍然是O(n)，并没有数量级上的优化。

希尔排序

在进行插入排序时，其最好情况的时间复杂度为O(n)，但是最坏，也就是逆序的情况下时间复杂度却是O(n^2)。所以如果可以调整数组，让其可以更接近最好情况，那么速度应该会更快。
希尔排序可以说是一种分成很多块的插入排序。在进行希尔排序时，先设定一个增量d，然后第一个分组就是{i,i+d,i+2d,…,i+nd}，第二个分组就是{(i+1)+d,(i+1)+2d,…,(i+1)+nd}，以此类推。之后对每一组进行插入排序，当所有组完成插入排序后d的值除以2，然后继续划分组、插入排序，直到d=1后进行一次整体的插入排序。

void ShellSort(int a[], int n)
{
	int i, j, tmp, d;
	for (d = n / 2; d >= 1; d /= 2)
	{
		for (i = d; i < n; i++)
		{
			if (a[i] < a[i - d])
			{
				tmp = a[i];
				for (j = i - d; j >= 0 && a[j] > tmp ; j-=d)
				{
					a[j + d] = a[j];
				}
				a[j + d] = tmp;
			}
		}
	}
}

冒泡排序

从后往前两个两个的进行比较，小的数就会往前走，最终最小的数就会冒到第一位，接着再从后往前冒，但不考虑已经冒上去的数。最终可以得到一个有序数组。
代码:

// 冒泡排序
void BubbleSort(int a[], int n)
{
	int tmp;
	int flag;
	for (int i = 0; i < n - 1; i++)
	{
		flag = 0;
		for (int j = n - 1; j > i; j--)
		{
			if (a[j] < a[j - 1])
			{
				tmp = a[j - 1];
				a[j - 1] = a[j];
				a[j] = tmp;
				flag = 1;
			}
		}
		if (flag == 0)
			return;
	}
}

快速排序

过程大致为。首先选定一个枢纽，选择第一元素作为枢纽，然后从最高位high开始一直往左检测，直到找到一个比枢纽小的元素，然后将high位置的元素移动到low位置，接着从low位置开始一直向右检测，直到找到比枢纽大的元素，然后low位置的元素移动到high位置，然后再对high进行相同操作，直至最终low=high，将枢纽放置在low=high的位置，此时第一次快速排序结束。
接着以枢纽为中心划分为了左右两个数组，此时直接调用QuickSort函数进行递归操作对左右两个数组进行排序，最终就可以得到一个整体的有序数组。
但是在递归过程中大量的调用了函数栈，这可能会导致栈的溢出。但也可以用非递归的方法实现快排。
代码如下:

// 快速排序
int Partition(int a[], int low, int high)
{
	int p = a[low];// 使需要排序的数组的第一个元素作为枢纽
	while (low < high)
	{
		// 一直使high减1，直到找到一个值大于枢纽
		while (low < high && a[high] >= p)high--;
		a[low] = a[high];// 将找到的大于枢纽的值赋予low位置
		// 同上
		while (low < high && a[low] <= p)low++;
		a[high] = a[low];
	}
	a[low] = p;
	return low;
}
void QuickSort(int a[], int low, int high)
{
	if (low < high)
	{
		int p = Partition(a, low, high);
		// 对枢纽左边进行快速排序
		QuickSort(a, low, p - 1);
		// 对枢纽右边进行快速排序
		QuickSort(a, p + 1, high);
	}
}

堆排序

这里把一个数组以二叉树的方式对待，从而建立大根堆
i 的左孩子 ————2i
i 的右孩子 ————2i+1
i 的父节点 ————[ i / 2 ]
检查当前节点是否满足 根>=左、右，若不满足，将当前节点与更大的一个孩子互换，若元素互换破环了下一级的堆，则采用相同的方法继续往下调整(小元素不断“下坠”);

// 堆排序
void HeadAdjust(int a[], int k, int len)
{
	int tmp = a[k];
	// i=2*k指向节点k的左孩子
	// i*=2 意为每次都重新指向新的孩子节点
	for (int i = 2 * k; i <= len; i*=2)
	{
		if (i < len && a[i] < a[i + 1])
			i++;// 筛选左孩子或者右孩子哪个值较大
		// 此时a[i]指向的节点为两个孩子之间较大值
		if (tmp >= a[i])break;// 满足大根堆，无需调整，跳出循环
		else
		{
			a[k] = a[i];// 将根节点设置为该子节点的较大值
			k = i;// 更新根节点
		}
	}
	a[k] = tmp;// 被筛选节点放入最终位置
}
void BuildMaxHeap(int a[], int len)
{
	for (int i = len / 2; i > 0; i--)
		HeadAdjust(a, i, len);
}
void HeapSort(int a[], int len)
{
	BuildMaxHeap(a, len);
	int tmp;
	for (int i = len; i > 1; i--)
	{
		tmp = a[i];
		a[i] = a[1];
		a[1] = tmp;
		HeadAdjust(a, 1, i - 1);
	}
}

// 直接插入排序
void InsertSort(int a[], int n)
{
	int i, j, temp; //temp是为了用来保存元素，防止原下标之前的元素在后移的过程中将该元素覆盖
	for (i = 1; i < n; i++)
	{
		if (a[i] < a[i - 1])// 如果当前元素比前一个元素要小，那么前面的元素就需要一直后移
		{
			int temp = a[i];
			for (j = i - 1; j >= 0; j--)
			{
				if (a[j] > temp)// 凡是大于temp的元素都需要后移
					a[j + 1] = a[j];
				else break;
			}
			a[j + 1] = temp;
		}
	}
}
// 随机数生成函数
int Rand(int x, int y)
{
	int b, d;
	if (y > x)
	{
		b = x;
		d = y - x + 1;
	}
	else
	{
		b = y;
		d = x - y + 1;
	}
	return (int)(1.0 * rand() * d / RAND_MAX + b);
}
// 折半插入排序
void HalfInsertSort(int a[], int n)
{
	int i, j, low, high, mid, tmp;
	for (i = 1; i < n; i++)
	{
		tmp = a[i];
		low = 0; high = i - 1;
		while (low <= high)
		{
			mid = (low + high) / 2;
			if (a[mid] > tmp)high = mid - 1;
			else low = mid + 1;
		}
		for (j = i - 1; j >= high + 1; j--)
			a[j + 1] = a[j];
		a[high + 1] = tmp;
	}
}
// 希尔排序
void ShellSort(int a[], int n)
{
	int i, j, tmp, d;
	for (d = n / 2; d >= 1; d /= 2)
	{
		for (i = d; i < n; i++)
		{
			if (a[i] < a[i - d])
			{
				tmp = a[i];
				for (j = i - d; j >= 0 && a[j] > tmp; j-=d)
				{
					a[j + d] = a[j];
				}
				a[j + d] = tmp;
			}
		}
	}
}
// 冒泡排序
void BubbleSort(int a[], int n)
{
	int tmp;
	int flag;
	for (int i = 0; i < n - 1; i++)
	{
		flag = 0;
		for (int j = n - 1; j > i; j--)
		{
			if (a[j] < a[j - 1])
			{
				tmp = a[j - 1];
				a[j - 1] = a[j];
				a[j] = tmp;
				flag = 1;
			}
		}
		if (flag == 0)
			return;
	}
}
// 快速排序
int Partition(int a[], int low, int high)
{
	int p = a[low];// 使需要排序的数组的第一个元素作为枢纽
	while (low < high)
	{
		// 一直使high减1，直到找到一个值大于枢纽
		while (low < high && a[high] >= p)high--;
		a[low] = a[high];// 将找到的大于枢纽的值赋予low位置
		// 同上
		while (low < high && a[low] <= p)low++;
		a[high] = a[low];
	}
	a[low] = p;
	return low;
}
void QuickSort(int a[], int low, int high)
{
	if (low < high)
	{
		int p = Partition(a, low, high);
		// 对枢纽左边进行快速排序
		QuickSort(a, low, p - 1);
		// 对枢纽右边进行快速排序
		QuickSort(a, p + 1, high);
	}
}
// 简单选择排序
void SelectSort(int a[], int n)
{
	for (int i = 0; i < n; i++)
	{
		for (int j = i + 1; j < n; j++)
		{
			if (a[i] > a[j])
			{
				int tmp = a[i];
				a[i] = a[j];
				a[j] = tmp;
			}
		}
	}
}
// 堆排序
void HeadAdjust(int a[], int k, int len)
{
	int tmp = a[k];
	// i=2*k指向节点k的左孩子
	// i*=2 意为每次都重新指向新的孩子节点
	for (int i = 2 * k; i <= len; i*=2)
	{
		if (i < len && a[i] < a[i + 1])
			i++;// 筛选左孩子或者右孩子哪个值较大
		// 此时a[i]指向的节点为两个孩子之间较大值
		if (tmp >= a[i])break;// 满足大根堆，无需调整，跳出循环
		else
		{
			a[k] = a[i];// 将根节点设置为该子节点的较大值
			k = i;// 更新根节点
		}
	}
	a[k] = tmp;// 被筛选节点放入最终位置
}
void BuildMaxHeap(int a[], int len)
{
	for (int i = len / 2; i > 0; i--)
		HeadAdjust(a, i, len);
}
void HeapSort(int a[], int len)
{
	BuildMaxHeap(a, len);
	int tmp;
	for (int i = len; i > 1; i--)
	{
		tmp = a[i];
		a[i] = a[1];
		a[1] = tmp;
		HeadAdjust(a, 1, i - 1);
	}
}
